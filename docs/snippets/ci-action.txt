jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Start LocalAI (docker-compose)
        run: |
          docker compose up -d
          timeout 120 bash -c 'until curl -s http://localhost:8080/v1/models | grep hermes-2-pro-mistral; do sleep 3; done'

      - name: Set up Python + CData MCP
        run: |
          sudo apt-get update && sudo apt-get install -y python3.12 python3.12-venv git curl
          python3.12 -m venv .venv
          source .venv/bin/activate
          pip install --upgrade pip uv
          git clone https://github.com/CDataSoftware/cdata-mcp-python.git
          cd cdata-mcp-python
          uv venv .venv
          source .venv/bin/activate
          pip install "mcp[cli]" ./cdata.postgresql.whl || echo "Attach your connector wheel here"
          export CONNECTOR_MOD="cdata.postgresql"
          export CONNECTION_STRING="Server=host.docker.internal;Port=5432;Database=mydb;User Id=user;Password=pass;"
          nohup uv run python -m mcp.server.fastmcp > mcp.log 2>&1 & sleep 5

      - name: Prepare .env
        run: |
          cp .env.example .env
          cat >> .env <<EOF
          MISTRAL_API_URL=http://localhost:8080/v1
          MISTRAL_API_KEY=localai-key
          LOCALAI_API_URL=http://localhost:8080/v1
          LOCALAI_API_KEY=localai-key
          MCP_SERVER_URL=http://127.0.0.1:8000
          EOF

      - name: Install node deps
        run: npm ci

      - name: Validate API keys
        run: node scripts/validate-api-keys.js

      - name: Run tests
        run: npm test
