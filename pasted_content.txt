    <script>
        // Initialize progress tracking
        let completedItems = 0;
        const totalItems = 56;

        document.addEventListener('DOMContentLoaded', function() {
            initializeProgressChart();
            loadProgress();
            updateProgress();
        });

        function initializeProgressChart() {
            const chartDom = document.getElementById('progressChart');
            const myChart = echarts.init(chartDom);
            
            const option = {
                tooltip: {
                    trigger: 'item',
                    formatter: '{a} <br/>{b}: {c} ({d}%)'
                },
                legend: {
                    orient: 'vertical',
                    left: 'left',
                    data: ['Completed', 'Remaining']
                },
                series: [
                    {
                        name: 'Progress',
                        type: 'pie',
                        radius: ['40%', '70%'],
                        center: ['60%', '50%'],
                        avoidLabelOverlap: false,
                        label: {
                            show: false,
                            position: 'center'
                        },
                        emphasis: {
                            label: {
                                show: true,
                                fontSize: '18',
                                fontWeight: 'bold'
                            }
                        },
                        labelLine: {
                            show: false
                        },
                        data: [
                            {
                                value: 0,
                                name: 'Completed',
                                itemStyle: { color: '#008DA8' }
                            },
                            {
                                value: totalItems,
                                name: 'Remaining',
                                itemStyle: { color: '#e5e7eb' }
                            }
                        ]
                    }
                ]
            };
            
            myChart.setOption(option);
            window.progressChart = myChart;
        }

        function updateProgress() {
            const checkboxes = document.querySelectorAll('.checklist-item input[type="checkbox"]');
            completedItems = Array.from(checkboxes).filter(cb => cb.checked).length;
            
            const percentage = Math.round((completedItems / totalItems) * 100);
            
            // Update progress ring
            const progressRing = document.getElementById('progressRing');
            const circumference = 2 * Math.PI * 15.9155;
            const strokeDasharray = (percentage / 100) * circumference;
            progressRing.style.strokeDasharray = `${strokeDasharray}, ${circumference}`;
            
            // Update text
            document.getElementById('progressPercent').textContent = `${percentage}%`;
            document.getElementById('completedCount').textContent = completedItems;
            
            // Update chart
            if (window.progressChart) {
                window.progressChart.setOption({
                    series: [{
                        data: [
                            {
                                value: completedItems,
                                name: 'Completed',
                                itemStyle: { color: '#008DA8' }
                            },
                            {
                                value: totalItems - completedItems,
                                name: 'Remaining',
                                itemStyle: { color: '#e5e7eb' }
                            }
                        ]
                    }]
                });
            }
            
            // Update category progress
            updateCategoryProgress();
            
            // Save progress
            saveProgress();
            
            // Check for completion
            if (completedItems === totalItems) {
                showCompletionCelebration();
            }
        }

        function updateCategoryProgress() {
            const categories = ['instance', 'project', 'user', 'integrations', 'advanced'];
            
            categories.forEach(category => {
                const categoryItems = document.querySelectorAll(`#${category}-content .checklist-item input[type="checkbox"]`);
                const completedCategoryItems = Array.from(categoryItems).filter(cb => cb.checked).length;
                const totalCategoryItems = categoryItems.length;
                
                const categoryProgress = document.querySelector(`#${category}-content`).previousElementSibling.querySelector('.category-progress');
                const categoryCompleted = document.querySelector(`#${category}-content`).previousElementSibling.querySelector('.category-completed');
                
                if (categoryProgress && categoryCompleted) {
                    const percentage = totalCategoryItems > 0 ? (completedCategoryItems / totalCategoryItems) * 100 : 0;
                    categoryProgress.style.width = `${percentage}%`;
                    categoryCompleted.textContent = completedCategoryItems;
                }
            });
        }

        function toggleCategory(categoryId) {
            const content = document.getElementById(`${categoryId}-content`);
            const button = event.target.closest('button');
            const icon = button.querySelector('svg');
            
            if (content.classList.contains('expanded')) {
                content.classList.remove('expanded');
                icon.style.transform = 'rotate(0deg)';
            } else {
                content.classList.add('expanded');
                icon.style.transform = 'rotate(180deg)';
            }
        }

        function markAllComplete() {
            if (confirm('Are you sure you want to mark all items as complete?')) {
                const checkboxes = document.querySelectorAll('.checklist-item input[type="checkbox"]');
                checkboxes.forEach(cb => cb.checked = true);
                updateProgress();
                showNotification('All items marked as complete!', 'success');
            }
        }

        function resetProgress() {
            if (confirm('Are you sure you want to reset all progress? This cannot be undone.')) {
                const checkboxes = document.querySelectorAll('.checklist-item input[type="checkbox"]');
                checkboxes.forEach(cb => cb.checked = false);
                updateProgress();
                showNotification('Progress reset successfully', 'info');
            }
        }

        function exportProgress() {
            const checkboxes = document.querySelectorAll('.checklist-item input[type="checkbox"]');
            const progressData = {
                completed: Array.from(checkboxes).filter(cb => cb.checked).length,
                total: totalItems,
                percentage: Math.round((completedItems / totalItems) * 100),
                completedAt: new Date().toISOString(),
                items: Array.from(checkboxes).map((cb, index) => ({
                    id: index,
                    completed: cb.checked,
                    text: cb.closest('.checklist-item').querySelector('.item-text').textContent
                }))
            };

            const dataStr = JSON.stringify(progressData, null, 2);
            const dataBlob = new Blob([dataStr], { type: 'application/json' });
            const url = URL.createObjectURL(dataBlob);
            
            const link = document.createElement('a');
            link.href = url;
            link.download = 'jira-setup-progress.json';
            link.click();
            
            URL.revokeObjectURL(url);
            showNotification('Progress report exported successfully!', 'success');
        }

        function saveProgress() {
            const checkboxes = document.querySelectorAll('.checklist-item input[type="checkbox"]');
            const progress = Array.from(checkboxes).map(cb => cb.checked);
            localStorage.setItem('jiraSetupProgress', JSON.stringify(progress));
        }

        function loadProgress() {
            const saved = localStorage.getItem('jiraSetupProgress');
            if (saved) {
                const progress = JSON.parse(saved);
                const checkboxes = document.querySelectorAll('.checklist-item input[type="checkbox"]');
                progress.forEach((isCompleted, index) => {
                    if (checkboxes[index]) {
                        checkboxes[index].checked = isCompleted;
                    }
                });
            }
        }

        function showCompletionCelebration() {
            const celebration = document.createElement('div');
            celebration.className = 'fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50';
            celebration.innerHTML = `
                <div class="bg-white rounded-2xl p-8 max-w-md mx-4 text-center">
                    <div class="w-16 h-16 bg-green-100 rounded-full flex items-center justify-center mx-auto mb-4">
                        <svg class="w-8 h-8 text-green-600" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                            <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z"></path>
                        </svg>
                    </div>
                    <h2 class="text-2xl font-bold text-gray-900 mb-2">Congratulations!</h2>
                    <p class="text-gray-600 mb-6">You've completed the Jira setup checklist. Your instance is ready for production use!</p>
                    <button onclick="this.closest('.fixed').remove()" class="px-6 py-3 bg-teal-600 text-white rounded-lg hover:bg-teal-700 transition-colors">
                        Continue to Jira
                    </button>
                </div>
            `;
            document.body.appendChild(celebration);

            // Animate celebration
            anime({
                targets: celebration.querySelector('.bg-white'),
                scale: [0.8, 1],
                opacity: [0, 1],
                duration: 500,
                easing: 'easeOutBack'
            });
        }

        function showNotification(message, type = 'info') {
            const notification = document.createElement('div');
            notification.className = `fixed top-20 right-4 z-50 p-4 rounded-lg shadow-lg max-w-sm ${
                type === 'success' ? 'bg-green-500 text-white' :
                type === 'warning' ? 'bg-yellow-500 text-white' :
                type === 'error' ? 'bg-red-500 text-white' :
                'bg-blue-500 text-white'
            }`;
            notification.textContent = message;
            
            document.body.appendChild(notification);
            
            anime({
                targets: notification,
                translateX: [300, 0],
                opacity: [0, 1],
                duration: 300,
                easing: 'easeOutQuad'
            });
            
            setTimeout(() => {
                anime({
                    targets: notification,
                    translateX: [0, 300],
                    opacity: [1, 0],
                    duration: 300,
                    easing: 'easeInQuad',
                    complete: () => notification.remove()
                });
            }, 4000);
        }
